$highest = Get-ChildItem -File 
    Where-Object { $_.BaseName -match '^d+$' } 
    ForEach-Object { [int]$_.BaseName } 
    Sort-Object -Descending 
    Select-Object -First 1

if (-not $highest) {
    $highest = 0
}

$count = $highest + 1

Get-ChildItem -File 
    Where-Object { $_.BaseName -notmatch '^d+$' } 
    Sort-Object Name 
    ForEach-Object {
        $newName = $count$($_.Extension)
        Rename-Item -Path $_.FullName -NewName $newName
        $count++
    }
